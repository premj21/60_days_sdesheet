#include <bits/stdc++.h> 
long long maxSubarraySum(int arr[], int n)
{    
      long long maxi = -1e9; 
    long long sum = 0 ; 
    for(int i = 0 ;i<n ; i++){
sum+=arr[i];
        if(sum<0) sum = 0 ;
    maxi = max(sum, maxi);
     
    }
    return maxi;
}
